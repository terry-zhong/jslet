    <script type="text/javascript">
        jslet.define("employee", "../js/employee.js");
		jslet.define("template", "jquery.tmpl.js");
        jslet.require("jslet,employee,template", initialzie);
		
		function initialzie(){
			//Dataset: employee is defined in employee.js
			//For initial field value list
			$( "#employeeTemplate" ).tmpl( jslet.data.getDataset('employee').dataList() )
				.appendTo( "#employeeDataList" );	
			
			//For converted field value list
			$( "#employeeTemplate" ).tmpl( jslet.data.getDataset('employee').textList() )
				.appendTo( "#employeeList" );		
		}
    </script>
