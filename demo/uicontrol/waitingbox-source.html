        function showWaiting1() {
            waitingbox = new jslet.ui.WaitingBox(document.body, "#CCCCCC");
            waitingbox.show("Please wait a moment(disappear after 2 seconds)...");
            setTimeout(function(){waitingbox.destroy(); waitingbox = null;}, 2000);
        }

        waitingbox1 = null;
        function showWaiting2() {
            if (!waitingbox1)
                waitingbox1 = new jslet.ui.WaitingBox(document.getElementById("test"), "Gray", true);
            waitingbox1.show("Please wait a moment...");
        }

        function hideWaiting2() {
            if (waitingbox1)
                waitingbox1.hide();
        }
