    <script type="text/javascript">
        jslet.define("salesdetail", "../js/salesdetail.js");
        jslet.require("jslet,salesdetail", function () {
            initialize();
        });

        function initialize() {
			dsSaleMaster = jslet.data.getDataset('dsSaleMaster');
            var idObj = dsSaleMaster.getField('saleid');
			idObj.editMask({mask:'LL-0000', keepChar: true, transform: 'upper'});

            var dateObj = dsSaleMaster.getField('saledate');
			dateObj.editMask({mask:'0000-00-00', keepChar: true});

			dsSaleMaster.dataList(null);
			dsSaleMaster.insertRecord();

	        jslet.ui.onReady = function(){
				dsSaleMaster.focusEditControl('saleid');
	        }
        }

    </script>
